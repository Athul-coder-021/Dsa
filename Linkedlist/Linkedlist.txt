--- CREATING THE START NODE ----

Node* head = new Node(arr[0]);
Node* current = head;

-- CREATING NEW LL FROM ARRAY --

for (int i = 1; i < arr.size(); i++) {
    current->next = new Node(arr[i]);
    current = current->next;
}
